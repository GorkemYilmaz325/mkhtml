185 // Registers function fn under the given name within the given RPC context
186 #define jsonrpc_ctx_export(ctx, name, fn) \
187  do { \
188  static struct jsonrpc_method m = {(name), sizeof(name) - 1, (fn), 0}; \
189  m.next = (ctx)->methods; \
190  (ctx)->methods = &m; \
191  } while (0)
